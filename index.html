// --- COMPOSANT IMAGE ÉDITORIALE (IMAGEN 4.0) ---

const EditorialImage = ({ prompt, className = "" }) => {

  const [imageUrl, setImageUrl] = useState(null);

  const [loading, setLoading] = useState(true);

  const [error, setError] = useState(false);


  useEffect(() => {

    let isMounted = true;

    

    const generateImage = async (retryCount = 0) => {

      try {

        const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/imagen-4.0-generate-001:predict?key=${apiKey}`, {

          method: 'POST',

          headers: { 'Content-Type': 'application/json' },

          body: JSON.stringify({

            instances: { prompt: `${prompt}, cinematic lighting, high-end architectural photography, luxury mediterranean pool, 8k resolution, ultra realistic` },

            parameters: { sampleCount: 1 }

          })

        });


        if (!response.ok) {

          if (retryCount < 5) {

            const delay = Math.pow(2, retryCount) * 1000;

            setTimeout(() => generateImage(retryCount + 1), delay);

            return;

          }

          throw new Error('Erreur API');

        }


        const result = await response.json();

        if (isMounted && result.predictions?.[0]?.bytesBase64Encoded) {

          setImageUrl(`data:image/png;base64,${result.predictions[0].bytesBase64Encoded}`);

          setError(false);

        } else {

          throw new Error('Pas de données');

        }

      } catch (e) {

        if (isMounted) setError(true);

      } finally {

        if (isMounted) setLoading(false);

      }

    };


    generateImage();

    return () => { isMounted = false; };

  }, [prompt]);


  return (

    <div className={`relative overflow-hidden bg-[#0A192F] flex items-center justify-center ${className}`}>

      {loading ? (

        <div className="flex flex-col items-center gap-3">

          <Loader2 className="w-6 h-6 animate-spin text-[#D4AF37]" />

          <span className="text-[9px] uppercase tracking-[0.3em] text-[#D4AF37]/50 font-bold">Rendu de l'image...</span>

        </div>

      ) : imageUrl ? (

        <img 

          src={imageUrl} 

          alt="Visuel de prestige" 

          className="w-full h-full object-cover animate-in fade-in zoom-in-105 duration-1000" 

        />

      ) : (

        <div className="flex flex-col items-center gap-2 opacity-20">

          <Waves size={40} />

          <span className="text-[8px] uppercase tracking-widest">Aix Services</span>

        </div>

      )}

    </div>

  );

};
